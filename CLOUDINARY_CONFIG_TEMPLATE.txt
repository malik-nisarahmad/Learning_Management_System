# Add these to your .env.local file
# You need CLOUD NAME (not API Key) and UPLOAD PRESET

# Step 1: Get Cloud Name from Cloudinary Dashboard
# Go to: https://console.cloudinary.com/
# Look at the top of the dashboard for your Cloud Name (e.g., "dxxxxx")

# Step 2: Create Upload Preset
# Go to: Settings → Upload → Upload presets → Add upload preset
# Name it: fast-connect-uploads
# Set to: Unsigned
# Save it

# Step 3: Add these values below:

NEXT_PUBLIC_CLOUDINARY_CLOUD_NAME=YOUR_CLOUD_NAME_HERE
NEXT_PUBLIC_CLOUDINARY_UPLOAD_PRESET=fast-connect-uploads

# Example (replace with your actual values):
# NEXT_PUBLIC_CLOUDINARY_CLOUD_NAME=dxxxxx
# NEXT_PUBLIC_CLOUDINARY_UPLOAD_PRESET=fast-connect-uploads

# IMPORTANT:
# - Cloud Name is NOT the API Key number
# - Cloud Name is usually a short string like "dxxxxx" or custom name
# - Upload Preset must be created in Cloudinary dashboard first
# - Restart server after adding these values
